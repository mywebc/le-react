### 安装

打开终端运行下列命令：

```js
npm install le-react-ui
// 或者
yarn add le-react-ui
```

### 引入组件

```js
import { Button, Switch, Radio } from "le-react-ui"
import "../node_modules/le-react-ui/dist/lib/leReactUI.css"
```

### 使用

```js
function App() {
  return (
    <div className="App">
        <Button>测试</Button>
    </div>
  );
}
```
